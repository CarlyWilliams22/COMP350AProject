//Some basic code

public class HelloWorld {

	/* 	This main method does not take any arguments.
		It creates a string of names and prints out a greeting to each.
	*/
	public static void main(String[] args) {
		if(true) {
			//create String array of guys
			String[] names = {"Billy", "Bob", "Joe", "Jeff"};
		} else {
			//create String array of girls
			String[] names = {"Lucy", "Lily", "Mary", "Martha"};
		}
		
		for(name : names) {
			System.out.println("Hello, " + name);
		} 

	}

}
